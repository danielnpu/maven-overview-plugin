228     private void generateOverview() {
229         getLog().debug(pluginName + " v" + pluginVersion + " build by " + pluginBuilder);
230         File outputFile = new File(getGraphLocation());
231         if (!outputFile.exists()) {
232             getLog().debug("Creating outputFile: " + outputFile.getAbsolutePath());
233             outputFile.getParentFile().mkdirs();
234             getLog().info("Created outputFile: " + outputFile);
235         }
236 
237         getLog().debug("Collecting data");
238         DependencyProcessor dependencyProcessor = new DependencyProcessor(
239                 excludes,
240                 dependencyTreeBuilder,
241                 localRepository,
242                 artifactFactory,
243                 artifactMetadataSource,
244                 artifactCollector,
245                 this);
246 
247         getLog().debug("Generating graph");
248         DirectedGraph graph = dependencyProcessor.createGraph(reactorProjects);
249 
250         getLog().debug("Rendering graph");
251         Layout layout = new KKLayout(graph);
252         Dimension preferredSize = new Dimension(width, height);
253         layout.initialize(preferredSize);
254         layout.resize(preferredSize);
255         if (layout.isIncremental()) {
256             getLog().info("Incrementing graph");
257             for (int i = 0; i < ITERATIONS; i++) {
258                 layout.advancePositions();
259             }
260         }
261 
262         PluggableRenderer renderer = setupRenderer();
263         VisualizationViewer viewer = new VisualizationViewer(layout, renderer,
264                 preferredSize);
265         viewer.setDoubleBuffered(false);
266         viewer.setSize(width, height);
267         viewer.setBackground(Color.WHITE);
268         Container container = new Container();
269         container.addNotify();
270         container.add(viewer);
271         container.setVisible(true);
272         BufferedImage image = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
273         Graphics2D graphics = image.createGraphics();
274 
275         container.paintComponents(graphics);
276 
277         getLog().debug("Writing image to " + outputFile.getAbsolutePath());
278         try {
279             ImageIO.write(image, "png", outputFile);
280             getLog().info("Graph at: " + outputFile);
281         } catch (IOException ioe) {
282             getLog().error("Couldn't write to: " + outputFile, ioe);
283         }
284     }

286     private PluggableRenderer setupRenderer() {
287         PluggableRenderer renderer = new PluggableRenderer();
288         renderer.setVertexStringer(new MyVertexStringer(vertexFullLabel));
289         renderer.setVertexPaintFunction(new MyVertexPaintFunction());
290         renderer.setVertexShapeFunction(new MyVertexShapeFunction());
291         renderer.setVertexLabelCentering(true);
292         renderer.setEdgeStringer(new MyEdgeStringer(suppressedScopes));
293         return renderer;
294     }
