33       * Etracting visitor.
34       */
35      protected class ExtractorDNV implements DependencyNodeVisitor {
36          private int depth = 0;
37          private DirectedGraph graph;
38          private Map<Artifact, ArtifactVertex> artifactVertexMap;
39          private Log log;
40  
41          /**
42           * Default constructor setting context for processing.
43           *
44           * @param graph             Graph to populate. In can be prepopulated already.
45           * @param artifactVertexMap Cache of all nodes in <code>graph</code>.
46           * @param log               Maven logger.
47           */
48          public ExtractorDNV(DirectedGraph graph, Map<Artifact, ArtifactVertex> artifactVertexMap, Log log) {
49              this.graph = graph;
50              this.artifactVertexMap = artifactVertexMap;
51              this.log = log;
52          }
53  
54          /**
55           * Starts the visit to the specified dependency node.
56           * <p/>
57           * Description
58           * <ul>
59           * <li>If vertex already in graph:
60           * <dl>
61           * <dt>Vertex not present in graph</dt>
62           * <dd>Creates new vertex in <code>graph</code>.</dd>
63           * <dt>Vertex already present in graph</dt>
64           * <dd>Adjusts distance of visited <code>node</code> if needed.</dd>
65           * </dl>
66           * </li>
67           * <li>If <code>node</code> has a parent, create edge connecting parent to visited node.</li>
68           * <li>Increases depth of visited dependency nodes.</li>
69           * <li>Allows for processing of child nodes.</li>
70           * </ul>
71           *
72           * @param node the dependency node to visit
73           * @return <code>true</code> to visit the specified dependency node's children, <code>false</code> to skip the
74           *         specified dependency node's children and proceed to its next sibling
75           */
76          public boolean visit(DependencyNode node) {
77              Artifact artifact = node.getArtifact();
78              ArtifactVertex vertex;
79              boolean wasHereBefore = false;
80              if (log.isDebugEnabled()) log.debug(depth + ": " + artifact.getId());
81              if (!artifactVertexMap.containsKey(artifact)) {
82                  // vertex not yet in graph
83                  vertex = new ArtifactVertex(artifact, depth);
84                  artifactVertexMap.put(artifact, vertex);
85                  graph.addVertex(vertex);
86              } else {
87                  // vertex already in graph align depth
88                  vertex = artifactVertexMap.get(artifact);
89                  vertex.alignDistance(depth);
90                  wasHereBefore = true;
91              }
92              if (node.getParent() != null) {
93                  // create edge connecting parent to visited node.
94                  final Artifact parentArtifact = node.getParent().getArtifact();
95                  graph.addEdge(
96                          new DependencyEdge(
97                                  artifactVertexMap.get(parentArtifact),
98                                  vertex,
99                                  artifact.getScope())
100                 );
101             }
102             depth++; // increment depth
103             return !wasHereBefore; // process child nodes if hasn't been here before
104         }
105 
106         /**
107          * Ends the visit to to the specified dependency node.
108          * <p/>
109          * Description: reduces depth of visited nodes, allows for child nodes processing.
110          *
111          * @param node the dependency node to visit
112          * @return <code>true</code> to visit the specified dependency node's next sibling, <code>false</code> to skip
113          *         the specified dependency node's next siblings and proceed to its parent
114          */
115         public boolean endVisit(DependencyNode node) {
116             depth--; // decrement depth
117             return true; // process child nodes
118         }
119     }

121     /**
122      * Extracts graph from dependency node.
123      *
124      * @param node              Node to start extracting from.
125      * @param graph             Populate this graph with extracted dependencies.
126      * @param artifactVertexMap Storage for (artifact, vertex) mapping.
127      * @param log               Maven logger.
128      */
129     public void extractGraph(DependencyNode node, DirectedGraph graph, Map<Artifact, ArtifactVertex> artifactVertexMap, Log log) {
130         node.accept(new ExtractorDNV(graph, artifactVertexMap, log));
131     }
