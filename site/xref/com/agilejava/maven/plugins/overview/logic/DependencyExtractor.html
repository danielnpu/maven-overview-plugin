34    /** Etracting visitor. */
35    protected class ExtractorDNV implements DependencyNodeVisitor {
36      /** Depth of execution. */
37      private int depth = 0;
38      /** Graph. */
39      private DirectedGraph graph;
40      /** Mapping of Artifacts to Vertexes. */
41      private Map<Artifact, ArtifactVertex> artifactVertexMap;
42      /**
43       * Artifac filter.
44       * <p/>
45       * TODO: Remove this if https://jira.codehaus.org/browse/MSHARED-4 gets
46       * fixed.
47       */
48      private MyArtifactFilter artifactFilter;
49      /** Logger. */
50      private Log log;
51  
52      /**
53       * Default constructor setting context for processing.
54       *
55       * @param directedGraph  Graph to populate. In can be prepopulated already.
56       * @param vertexMap      Cache of all nodes in <code>graph</code>.
57       * @param artifactFilter Artifac filter. Remove this when https://jira.codehaus.org/browse/MSHARED-4
58       *                       get's fixed.
59       * @param logger         Maven logger.
60       */
61      public ExtractorDNV(
62          final DirectedGraph directedGraph,
63          final Map<Artifact, ArtifactVertex> vertexMap,
64          final MyArtifactFilter artifactFilter,
65          final Log logger) {
66        this.graph = directedGraph;
67        this.artifactVertexMap = vertexMap;
68        this.artifactFilter = artifactFilter;
69        this.log = logger;
70      }
71  
72      /**
73       * Starts the visit to the specified dependency node.
74       * <p/>
75       * Description <ul> <li>If vertex already in graph: <dl> <dt>Vertex not
76       * present in graph</dt> <dd>Creates new vertex in <code>graph</code>.</dd>
77       * <dt>Vertex already present in graph</dt> <dd>Adjusts distance of visited
78       * <code>node</code> if needed.</dd> </dl> </li> <li>If <code>node</code>
79       * has a parent, create edge connecting parent to visited node.</li>
80       * <li>Increases depth of visited dependency nodes.</li> <li>Allows for
81       * processing of child nodes.</li> </ul>
82       *
83       * @param node the dependency node to visit
84       * @return <code>true</code> to visit the specified dependency node's
85       *         children, <code>false</code> to skip the specified dependency
86       *         node's children and proceed to its next sibling
87       */
88      public final boolean visit(final DependencyNode node) {
89        Artifact artifact = node.getArtifact();
90        if (!artifactFilter.include(artifact)) {
91          depth++;
92          return false;
93        }
94        ArtifactVertex vertex;
95        boolean wasHereBefore = false;
96  
97        if (log.isDebugEnabled()) {
98          log.debug("ExtractorDNV: " + depth + ": " + artifact.getId());
99        }
100 
101       if (!artifactVertexMap.containsKey(artifact)) {
102         // vertex not yet in graph
103         vertex = new ArtifactVertex(artifact, depth);
104         artifactVertexMap.put(artifact, vertex);
105         graph.addVertex(vertex);
106         if (log.isDebugEnabled()) {
107           log.debug("ExtractorDNV: " + "added vertex.");
108         }
109       } else {
110         // vertex already in graph align depth
111         vertex = artifactVertexMap.get(artifact);
112         vertex.alignDistance(depth);
113         wasHereBefore = true;
114         if (log.isDebugEnabled()) {
115           log.debug("ExtractorDNV: " + "aligned vertex.");
116         }
117       }
118       final DependencyNode parentNode = node.getParent();
119       final DependencyEdge dependencyEdge;
120       if (parentNode != null) {
121         // create edge connecting parent to visited node.
122         final Artifact parentArtifact = parentNode.getArtifact();
123         dependencyEdge = new DependencyEdge(
124             artifactVertexMap.get(parentArtifact),
125             vertex,
126             artifact.getScope());
127 
128         if (log.isDebugEnabled()) {
129           log.debug(
130               "ExtractorDNV: " + "adding edge from parent: "
131               + parentArtifact.getId() + ".");
132         }
133         try {
134           graph.addEdge(dependencyEdge);
135         } catch (ConstraintViolationException e) {
136           /* (2) If the constraint is not simple (as in the example above;
137           NotPredicate takes a Predicate as an argument), catch the
138           ConstraintViolationException, call getViolatedConstraint() on it, and
139           use PredicateUtils.evaluateNestedPredicates() on the constraint to
140           find out the results of evaluating each constituent predicate.
141           */
142           Map result = PredicateUtils.evaluateNestedPredicates(
143               e.getViolatedConstraint(), dependencyEdge);
144           for (Object entry : result.entrySet()) {
145             log.warn(
146                 "Predicate Violation: key: " + ((Map.Entry) entry).getKey()
147                 + ", value: " + ((Map.Entry) entry).getValue());
148           }
149         }
150       } else {
151         if (log.isDebugEnabled()) {
152           log.debug("ExtractorDNV: was here before: " + wasHereBefore + ".");
153           if (parentNode != null) {
154             log.debug(
155                 "ExtractorDNV: parenNode: "
156                 + parentNode.getArtifact().getId() + ".");
157           }
158         }
159       }
160 
161       depth++; // increment depth
162       return !wasHereBefore; // process child nodes if hasn't been here before
163     }
164 
165     /**
166      * Ends the visit to to the specified dependency node.
167      * <p/>
168      * Description: reduces depth of visited nodes, allows for child nodes
169      * processing.
170      *
171      * @param node the dependency node to visit
172      * @return <code>true</code> to visit the specified dependency node's next
173      *         sibling, <code>false</code> to skip the specified dependency
174      *         node's next siblings and proceed to its parent
175      */
176     public final boolean endVisit(final DependencyNode node) {
177       depth--; // decrement depth
178       return true; // process next sibling
179     }
180   }
