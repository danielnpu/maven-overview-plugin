10  /**
11   * Artifact filtering.
12   * <p/>
13   * Configured by two lists:
14   * <dl>
15   * <dt>includes</dt><dd>List of included <em>groupID</em>s.</dd>
16   * <dt>excludes</dt><dd>List of excluded <em>ID</em>s of artifacts.</dd>
17   * </dl>
18   * <p/>
19   * Behavior (<code color="green">true</code> artifact included,
20   * <code color="red">false</code> artifact filtered out):<br/>
21   * if <code>includes</code> present:
22   * <dl>
23   * <dt>true</dt><dd>if artifacts groupID is found in <code>includes</code> then
24   * <code color="green">true</code>, else <code color="red">false</code>.</dd>
25   * <dt>false</dt>
26   * <dd>if <code>excludes</code> present:
27   * <dl>
28   * <dt>true</dt><dd>if artifacts ID found in <code>excludes</code> than
29   * <code color="red">false</code>, else <code color="green">true</code>.</dd>
30   * <dt>false</dt><dd><code color="green">true</code>.</dd>
31   * </dl>
32   * </dd>
33   * </dl>
34   * <p/>
35   * TODO: due to https://jira.codehaus.org/browse/MSHARED-4 filtering broken.
36   *
37   * @author <a href="mailto:Hubert.Iwaniuk@gmail.com">Hubert Iwaniuk</a>
38   */
39  class MyArtifactFilter implements ArtifactFilter {
40  
41    /**
42     * List of excluded artifactIDs.
43     */
44    private List<String> excludes;
45    /**
46     * List of included groupIDs.
47     */
48    private List<String> includes;
49  
50    /**
51     * Ctor setting finltering parameters.
52     *
53     * @param includes List of included groupIDs.
54     * @param excludes List of excluded artifactIDs.
55     * @param log      logger.
56     */
57    public MyArtifactFilter(
58        final List < String > includes, List < String > excludes, final Log log) {
59      this.excludes = excludes != null ? new ArrayList < String > (excludes) : null;
60      this.includes = includes != null ? new ArrayList <String> (includes) : null;
61      log.debug(
62          "MyArtifactFilter: includes: \'" + includes + "\'.");
63      log.debug(
64          "MyArtifactFilter: excludes: \'" + excludes + "\'.");
65    }
66  
67    /**
68     * {@inheritDoc}
69     */
70    public boolean include(Artifact artifact) {
71      if (includes != null && !includes.isEmpty()) {
72        boolean incl = false;
73        for (String include : includes) {
74          if (artifact.getGroupId().startsWith(include)) {
75            incl = true;
76            break;
77          }
78        }
79        return incl;
80      } else
81        return excludes == null || excludes.isEmpty() || !excludes.contains(
82            artifact.getId());
83    }
